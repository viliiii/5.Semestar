//4 boda točan, -0.8 bodova netočan, 0 bodova neodgovoren
//Što će ispisati sljedeći program? Ako je neka vrijednost nema isto ponašanje u svim slučajevima(broj dretvi)
//onda se ispis varijable definira kao ND

int main(){
    int v1 = 5; int v2 = 10; int v3 = 15; int v4 = 20;
    omp_set_num_threads(3);
    #pragma omp parallel shared(v1, v2) firstprivate(v3,v4)
    {
        v1++; v2++; v3++; v4++;
    } 
    printf("v1 = %d\nv2 = %d\nv3 = %d\nv4 = %d", v1, v2, v3, v4);
}

//Broj dretvi je 2

int main(){
    int v1 = 10; int v2 = 20;
    omp_set_num_threads(2);
    #pragma omp parallel shared(v1, v2)
    {
        #pragma omp single firstprivate(v2)
        {
            v1++; v2++;
        }
    }
    printf("v1=%d v2=%d", v1, v2);
}
